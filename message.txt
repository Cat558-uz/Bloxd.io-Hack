                    inspectTooltip.style = "position:fixed; background:#000; color:#fff; border:1px solid #fff; padding:4px 8px; font-size:11px; z-index:10000001; pointer-events:none; font-family:monospace; border-radius:4px;";
                    document.body.appendChild(inspectTooltip);
                    
                    document.addEventListener("mousemove", updateInspectTooltip);
                    document.addEventListener("click", handleInspectClick, true);
                    showNotification("Clique em um elemento para abrir o código!");
                }
            });

            addButton("Abrir Executer (Console)", secTools, () => {
                createExecuterWindow();
                showNotification("Executer Iniciado");
            });

            // --- RECURSOS ÚTEIS ADICIONAIS ---
            const secUtilitarios = addSection("Utilitários", content);

            addButton("Forçar Destravar Mouse", secUtilitarios, () => {
                // Remove o bloqueio de ponteiro (PointerLock) caso o jogo congele sua câmera
                document.exitPointerLock = document.exitPointerLock || document.mozExitPointerLock || document.webkitExitPointerLock;
                if (document.exitPointerLock) {
                    document.exitPointerLock();
                    showNotification("Cursor Destravado");
                }
            });
        }

        if (state.tab === "main") {
            const secFiltros = addSection("Controle de Tela", content);
            
            addButton("Inverter Cromatização (Modo Dark)", secFiltros, () => {
                const html = document.documentElement;
                html.style.filter = html.style.filter === "invert(1)" ? "none" : "invert(1)";
                showNotification("Visual Alterado");
            });

            addSlider("Opacidade Menu", 30, 100, 98, secFiltros, (val) => {
                root.style.setProperty('--bg-main', `rgba(10, 10, 10, ${val / 100})`);
            });
        }
    };

    // --- Escuta de Teclas para o HUD WASD Estilizado ---
    document.addEventListener("keydown", (e) => {
        const key = e.key.toUpperCase();
        if (["W", "A", "S", "D"].includes(key)) {
            const el = document.getElementById(`wasd-${key}`);
            if (el) {
                el.style.background = "#ffffff";
                el.style.color = "#000000";
            }
        }
    });

    document.addEventListener("keyup", (e) => {
        const key = e.key.toUpperCase();
        if (["W", "A", "S", "D"].includes(key)) {
            const el = document.getElementById(`wasd-${key}`);
            if (el) {
                el.style.background = "rgba(255,255,255,0.04)";
                el.style.color = "var(--text-dim)";
            }
        }
    });

    // --- Controle de Abas e Janela ---
    tabs.forEach(t => {
        t.onclick = () => {
            tabs.forEach(x => x.classList.remove("active"));
            t.classList.add("active");
            state.tab = t.dataset.tab;
            render();
        };
    });

    root.querySelector("#min-btn").onclick = () => {
        state.minimized = !state.minimized;
        root.querySelector("#ui-body").style.display = state.minimized ? "none" : "flex";
        root.querySelector("#min-btn").textContent = state.minimized ? "+" : "−";
    };

    root.querySelector("#close-btn").onclick = () => {
        root.style.opacity = "0";
        root.style.transform = "scale(0.95)";
        setTimeout(() => root.remove(), 150);
    };

    // --- Sistema de Arrastar (Drag) ---
    let drag = false, ox = 0, oy = 0;
    header.onmousedown = (e) => {
        if (e.target.classList.contains('ctl-btn')) return;
        drag = true;
        ox = e.clientX - root.offsetLeft;
        oy = e.clientY - root.offsetTop;
    };
    document.addEventListener("mousemove", (e) => { if (drag) { root.style.left = `${e.clientX - ox}px`; root.style.top = `${e.clientY - oy}px`; } });
    document.addEventListener("mouseup", () => drag = false);

    // --- Redimensionamento (Resize) ---
    let resizing = false, startW = 0, startH = 0, startX = 0, startY = 0;
    root.querySelector("#ui-resizer").onmousedown = (e) => {
        e.preventDefault(); e.stopPropagation();
        resizing = true;
        startW = root.clientWidth; startH = root.clientHeight;
        startX = e.clientX; startY = e.clientY;
    };
    document.addEventListener("mousemove", (e) => {
        if (!resizing) return;
        root.style.width = `${startW + (e.clientX - startX)}px`;
        root.style.height = `${startH + (e.clientY - startY)}px`;
    });
    document.addEventListener("mouseup", () => resizing = false);

    // --- Verificação de Versão Limpa ---
    const checkUpdates = () => {
        if (typeof GM_xmlhttpRequest === "undefined") return;
        GM_xmlhttpRequest({
            method: "GET",
            url: UPDATE_URL,
            onload: (res) => {
                const match = res.responseText.match(/version\s*([\d.]+)/i);
                if (match && match[1]) {
                    const remoteVersion = match[1].trim();
                    if (remoteVersion !== CURRENT_VERSION) {
                        showNotification(`Update disponivel: v${remoteVersion}`);
                    }
                }
            }
        });
    };

    render();
    checkUpdates();
})();
